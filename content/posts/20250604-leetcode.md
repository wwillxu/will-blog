## 31. Next Permutation

```java
class Solution {
    public void nextPermutation(int[] nums) {
        if (nums == null || nums.length == 1) {
            return;
        }

        int n = nums.length;

        // find first decreased num
        int i = n - 2;
        while (i >= 0 && nums[i] >= nums[i + 1]) {
            i--;
        }

        // if i exist, swap with the last bigger num
        // and reverse i+1 to end
        if (i >= 0) {
            int j = n - 1;
            while (j > i && nums[j] <= nums[i]) {
                j--;
            }
            swap(nums, i, j);
            reverse(nums, i + 1, n - 1);
            return;
        }
        // if i not found, reverse nums
        else {
            reverse(nums, 0, n - 1);
        }
    }

    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }

    private void reverse(int[] nums, int i, int j) {
        while (i < j) {
            swap(nums, i++, j--);
        }
    }
}
```



## 34. Find First and Last Position of Element in Sorted Array

```java
class Solution {
    public int[] searchRange(int[] nums, int target) {
        int[] res = new int[] { -1, -1 };

        if (nums == null || nums.length == 0) {
            return res;
        }

        // find left bound
        res[0] = findBound(nums, target, true);
        if (res[0] == -1) {
            return res;
        }

        // find right bound
        res[1] = findBound(nums, target, false);

        return res;
    }


    private int findBound(int[] nums, int target, boolean findLeft) {
        int left = 0, right = nums.length - 1;
        int mid = 0;
        int res = -1;

        while (left <= right) {
            mid = left + (right - left) / 2;
            if (nums[mid] == target) {
                res = mid;
                if (findLeft) {
                    right = mid - 1;
                } else {
                    left = mid + 1;
                }
            } else if (nums[mid] < target) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }

        return res;
    }
}
```
